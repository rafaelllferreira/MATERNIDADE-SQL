CREATE TABLE MAE(
    CPF INTEGER NOT NULL,
    NOME VARCHAR(50),
    ENDERECO VARCHAR(100),
    TELEFONE VARCHAR(11),
    PRIMARY KEY(CPF)
    ); 

CREATE TABLE EQUIPE_MEDICA(
    CODIGO_EQUIPE INTEGER NOT NULL,
    NOME_EQUIPE VARCHAR(20),
    PRIMARY KEY(CODIGO_EQUIPE)
    );

CREATE TABLE BEBE(
    CODIGO INTEGER NOT NULL,
    NOME VARCHAR(100),
    PESO FLOAT,
    ALTURA FLOAT,
    DATA_NASCIMENTO DATE,
    CPF INTEGER,
    CODIGO_EQUIPE INTEGER,
    PRIMARY KEY (CODIGO),
    FOREIGN KEY (CPF) REFERENCES mae (CPF),
    FOREIGN KEY (CODIGO_EQUIPE) REFERENCES equipe_medica (CODIGO_EQUIPE)
    );

CREATE TABLE PROFISSIONAL(
    MATRICULA INTEGER NOT NULL,
    NOME VARCHAR(20),
    ESPECIALIDADE VARCHAR(20),
    CRM INTEGER,
    TELEFONE VARCHAR(11),
    CODIGO_EQUIPE INTEGER,
    PRIMARY KEY(MATRICULA),
    FOREIGN KEY (CODIGO_EQUIPE) REFERENCES equipe_medica (CODIGO_EQUIPE)
    );

CREATE TABLE PACIENTE(
    CPF INTEGER NOT NULL,
    NOME VARCHAR(50),
    ENDERECO VARCHAR(100),
    PRIMARY KEY(CPF)
    ); 

CREATE TABLE MEDICO(
    CRM INTEGER NOT NULL,
    NOME VARCHAR(20),
    ESPECIALIDADE VARCHAR(20),
    PRIMARY KEY(CRM)
    );
CREATE TABLE TIPO_EXAME(
    NUMERO_EXAME INTEGER NOT NULL,
    TIPO_EXAME VARCHAR(20),
    CONVENIO VARCHAR(20),
    REQUISITOS VARCHAR(100),
    VALOR INTEGER,
    PRIMARY KEY(NUMERO_EXAME)
    );

CREATE TABLE EXAME(
    ID_EXAME INTEGER NOT NULL,
    DATA INTEGER,
    CPF INTEGER,
    NUMERO_EXAME INTEGER,
    PRIMARY KEY(ID_EXAME)
    FOREIGN KEY (NUMERO_EXAME) REFERENCES tipo_exame (NUMERO_EXAME)
    );

CREATE TABLE PRONTUARIO(
    NUMERO INTEGER NOT NULL,
    CRM INTEGER,
    CPF INTEGER,
    PRESCRIÇÃO VARCHAR(100),
    DATA DATE,
    PRIMARY KEY(NUMERO),
    FOREIGN KEY (CRM) REFERENCES medico (CRM),
    FOREIGN KEY (CPF) REFERENCES paciente (CPF)
    );
